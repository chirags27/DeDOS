483 static void parse_cmd_action(char *cmd) {
484  size_t ln = strlen(cmd) - 1;
485  if (*cmd && cmd[ln] == '\n') {
486  cmd[ln] = '\0';
487  }
488 
489  char cmd_cpy[256];
490  strcpy(cmd_cpy, cmd);
491 
492  int rtn = -1;
493  for ( int i = 0; cmd_actions[i].cmd[0] != '\0'; i++ ){
494  if ( strncasecmp(cmd, cmd_actions[i].cmd, strlen(cmd_actions[i].cmd)) == 0 ) {
495  if ( cmd_actions[i].action ) {
496  rtn = cmd_actions[i].action(cmd + strlen(cmd_actions[i].cmd) + 1 );
497  } else {
498  rtn = 1;
499  }
500  break;
501  }
502  }
503  if (rtn < 0){
504  log_error("Error parsing command: %s", cmd_cpy);
505  }
506 
507  return;
508 }
