503 static int find_time_index(struct timed_stat *stats, struct timespec *time,
504  int start_index, int stat_size) {
505  if (start_index < 0) {
506  start_index = stat_size - 1;

714 int init_stat_item(enum stat_id stat_id, unsigned int item_id) {
715  CHECK_INITIALIZATION;
716 
717  struct stat_type *type = &stat_types[stat_id];
718  if (!type->enabled) {
719  log(LOG_STATS, "Skipping initialization of type %s item %d",
720  type->label, item_id);
721  return 0;
722  }
723  if (wrlock_type(type) != 0) {
724  return -1;
725  }
726  if (type->id_indices[item_id] != -1) {
727  log_error("Item ID %u already assigned index %d for stat %s",
728  item_id, type->id_indices[item_id], type->label);
729  return -1;
730  }
731  int index;
732  for (index=0; index<type->max_items; index++) {
733  if (type->items[index].id == -1) {
734  break;
735  }
736  }
737  if (index == type->max_items) {
738  type->max_items++;
739  type->items = realloc(type->items, sizeof(*type->items) * (type->max_items));
740  realloc_stat_samples(stat_id, type->max_items);
741 
742  if ( type->items == NULL ) {
743  log_error("Error reallocating stat item");
744  return -1;
745  }
746  }
747  log(LOG_STATS, "Item %u has index %d", item_id, index);
748  struct stat_item *item = &type->items[index];
749 
750  if (pthread_mutex_init(&item->mutex,NULL)) {
751  log_error("Error initializing item %u lock", item_id);
752  return -1;
753  }
754 
755  item->id = item_id;
756  item->write_index = 0;
757  item->rolled_over = false;
758  item->stats = calloc(type->max_stats, sizeof(*item->stats));
759 
760  if (item->stats == NULL) {
761  log_error("Error calloc'ing item statistics");
762  return -1;
763  }
764 
765  type->id_indices[item_id] = index;
766 
767  type->num_items++;
768 
769  if (unlock_type(type) != 0) {
770  return -1;
771  }
772 
773  log(LOG_STAT_INITS, "Initialized stat item %s.%d (idx: %d)",
774  stat_types[stat_id].label, item_id, index);
775  return 0;
776 }
