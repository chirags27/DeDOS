41 static int handle_read(struct read_state *read_state,
42  struct ws_read_state *msu_state,
43  struct local_msu *self,
44  struct msu_msg *msg) {
45  log(LOG_WEBSERVER_READ, "Attempting read from %p", read_state);
46  int rtn = read_request(read_state);
47  if (rtn & (WS_INCOMPLETE_WRITE | WS_INCOMPLETE_READ)) {
48  read_state->conn.status = CON_READING;
49  log(LOG_WEBSERVER_READ, "Read incomplete. Re-enabling (fd: %d)", read_state->conn.fd);
50  msu_monitor_fd(read_state->conn.fd, RTN_TO_EVT(rtn), self, &msg->hdr);
51  free(msg->data);
52  return 0;
53  } else if (rtn & WS_ERROR) {
54  close_connection(&read_state->conn);
55  read_state->req_len = -1;
56  // Send the failure message (req_len = -1) to http
57  } else {
58  log(LOG_WEBSERVER_READ, "Read %s", read_state->req);
59  }
60  struct read_state *out = malloc(sizeof(*out));
61  memcpy(out, read_state, sizeof(*read_state));
62  msu_free_state(self, &msg->hdr.key);
63  free(msg->data);
64  call_msu_type(self, &WEBSERVER_HTTP_MSU_TYPE, &msg->hdr, sizeof(*out), out);
65  return 0;
66 }

68 static int handle_accept(struct read_state *read_state,
69  struct ws_read_state *msu_state,
70  struct local_msu *self,
71  struct msu_msg *msg) {
72  int rtn = accept_connection(&read_state->conn, msu_state->use_ssl);
73  if (rtn & WS_COMPLETE) {
74  rtn = handle_read(read_state, msu_state, self, msg);
75  return rtn;
76  } else if (rtn & WS_ERROR) {
77  close_connection(&read_state->conn);
78  msu_free_state(self, &msg->hdr.key);
79  free(msg->data);
80  return -1;
81  } else {
82  read_state->conn.status = CON_SSL_CONNECTING;
83  return msu_monitor_fd(read_state->conn.fd, RTN_TO_EVT(rtn), self, &msg->hdr);
84  }
85 }

87 static int read_http_request(struct local_msu *self,
88  struct msu_msg *msg) {
89  struct ws_read_state *msu_state = self->msu_state;
90 
91  struct connection conn_in;
92  int sender_type_id = msu_msg_sender_type(&msg->hdr.provinance);
93  switch (sender_type_id) {
94  case SOCKET_MSU_TYPE_ID:;
95  struct socket_msg *msg_in = msg->data;
96  init_connection(&conn_in, msg_in->fd);
97  break;
98  case WEBSERVER_HTTP_MSU_TYPE_ID:
99  log(LOG_WEBSERVER_READ, "Got back read msg %u", msg->hdr.key.id);
100  conn_in = *(struct connection*)msg->data;
101  break;
102  default:
103  log_error("Unknown sender MSU type ID: %d", sender_type_id);
104  return -1;
105  }
106 
107  size_t size = -1;
108  struct read_state *read_state = msu_get_state(self, &msg->hdr.key, &size);
109  if (read_state == NULL) {
110  read_state = msu_init_state(self, &msg->hdr.key, sizeof(*read_state));
111  init_read_state(read_state, &conn_in);
112  } else {
113  log(LOG_WEBSERVER_READ, "Retrieved read ptr %p", read_state);
114  }
115  if (read_state->conn.fd != conn_in.fd) {
116  log_error("Got non-matching FDs! state: %d vs input: %d", read_state->conn.fd, conn_in.fd);
117  return -1;
118  }
119 
120  switch (read_state->conn.status) {
121  case NIL:
122  case NO_CONNECTION:
123  case CON_ACCEPTED:
124  case CON_SSL_CONNECTING:
125  return handle_accept(read_state, msu_state, self, msg);
126  case CON_READING:
127  // TODO: I think it may never get here...
128  return handle_read(read_state, msu_state, self, msg);
129  default:
130  log_error("Received unknown packet status: %d", read_state->conn.status);
131  return -1;
132  }
133 }
